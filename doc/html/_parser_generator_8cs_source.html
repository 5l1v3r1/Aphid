147  private string GenerateLexer(List<AphidExpression> nodes, string code)
148  {
149  var lexerCall = nodes.SingleOrDefault(x =>
150  x.Type == AphidExpressionType.CallExpression &&
151  x.ToCall().FunctionExpression.Type == AphidExpressionType.IdentifierExpression &&
152  x.ToCall().FunctionExpression.ToIdentifier().Identifier == "Lexer");
153 
154  if (lexerCall == null)
155  {
156  throw new InvalidOperationException();
157  }
158 
159  var lexerObj = lexerCall.ToCall().Args.SingleOrDefault() as ObjectExpression;
160 
161  if (lexerObj == null)
162  {
163  throw new InvalidOperationException();
164  }
165 
166  nodes.Remove(lexerCall);
167  var ast = new List<AphidExpression>();
168 
169  var initKvp = lexerObj.Pairs.SingleOrDefault(x =>
170  x.LeftOperand.Type == AphidExpressionType.IdentifierExpression &&
171  x.LeftOperand.ToIdentifier().Identifier == "init");
172 
173  if (initKvp != null)
174  {
175  var initFunc = initKvp.RightOperand as FunctionExpression;
176 
177  if (initFunc == null)
178  {
179  throw new InvalidOperationException();
180  }
181 
182  ast.AddRange(initFunc.Body);
183  }
184 
185  ast.Add(new UnaryOperatorExpression(AphidTokenType.retKeyword, lexerObj)
186  .WithPositionFrom(lexerObj));
187 
188  var interpreter = new AphidInterpreter();
189  interpreter.CurrentScope.Add("nodes", new AphidObject(nodes));
190  interpreter.Interpret(ast);
191  var lexerCode = AlxFile.From(interpreter.GetReturnValue());
192 
193  var tokenTypeEnum = Regex.Match(
194  lexerCode,
195  @"public\s+enum\s+" + _config.TokenType + @"(.|\s)*?\{((.|\s)*?)\}");
196 
197  if (!tokenTypeEnum.Success)
198  {
199  throw new InvalidOperationException(string.Format(
200  "[Parser Generation] Could not find token type: {0}",
201  _config.TokenType));
202  }
203 
204  _tokenTypes = tokenTypeEnum.Groups[2].Value
205  .Split(new[] { '\r', '\n', ',', }, StringSplitOptions.RemoveEmptyEntries)
206  .Select(x => x.Trim())
207  .Where(x => x.Any())
208  .ToArray();
209 
210  return lexerCode;
211  }
